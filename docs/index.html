@magic/log

nodejs console.log with environment aware loglevels.

getting started

be in a nodejs project.

install

npm i --save @magic/error

import

import log from '@magic/log'

usage

import error from '@magic/error'err = error('message', 'name')// Error { message: 'message', name: 'E_NAME', code: 'E_NAME' }err = error('message')// Error { message: 'message', name: 'E_UNKNOWN', code: 'E_UNKNOWN' }err = error(new Error('message'))// Error { message: 'message', name: 'Error', code: 'E_UNKNOWN' }err = error('message', 'name with spaces')// Error { message: 'message', name: 'name with spaces', code: 'E_NAME_WITH_SPACES' }throw err// the stacktrace will be as expected,// not including the file in this lib that this error was instantiated in.
